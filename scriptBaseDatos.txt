mysql -u root -p

CREATE DATABASE RENTAS;

USE RENTAS;

CREATE TABLE RENTA
(
REN_FECHA varchar(50) NOT NULL,
REN_NOMBRE varchar(50) NOT NULL,
REN_IDENTIFICACION varchar(50) NOT NULL PRIMARY KEY,
REN_TELEFONO varchar(20) NULL,
REN_PLACA varchar(6) NOT NULL,
REN_COSTO INTEGER NOT NULL
);

CREATE TABLE CLIENTES
(
CL_NOMBRE varchar(50) NOT NULL,
CL_DIRECCION varchar(100) NOT NULL,
CL_CIUDAD varchar(50) NOT NULL,
CL_IDENTIFICACION varchar(20) NOT NULL PRIMARY KEY,
CL_LICENCIA varchar(20) NOT NULL,
CL_TELEFONO varchar(20) NULL
);

CREATE TABLE CARROS
(
CAR_PLACA varchar(6) NOT NULL PRIMARY KEY,
CAR_CONSUMO varchar(20) NOT NULL,
CAR_PUESTOS varchar(5) NOT NULL,
CAR_COLOR varchar(50) NOT NULL,
CAR_MARCA varchar(50) NOT NULL,
CAR_REFERENCIA varchar(80) NOT NULL,
CAR_VALOR varchar(50) NOT NULL
);
